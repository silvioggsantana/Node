instalar pacote : 
    npm install (pacote)

salvar dependências dev: 
    --save-dev

atualização de pacotes: 
    npm update 

verificar atualização:  
    npx npm=check-updates -up

scripts: {
    start : (comando que deseja rodar)
}

pacote global: 
    npm install (pacote) -g
    npm link (pacote)

executar pacote: 
    npm cowsay 'string'

remover pacote: 
    npm uninstall (pacote)
